00086     StreamTokenizer::TokenType token_type = st.PeekTokenType();
00087     if (token_type != StreamTokenizer::IDENTIFIER) {
00088       ostringstream err_ss;

00213 class Initializer<vector<T> > : public StreamInitializer {
00214  public:
00215   Initializer(vector<T> *member) : member_(member) { }
00216   virtual void Init(StreamTokenizer &st, Environment *env = NULL) {
00217     // Either the next token is an open brace (if reading tokens from
00218     // within a Factory-constructible object's member init list), or
00219     // else we just read an open brace (if Interpreter is reading tokens).
00220     if (st.Peek() == "{") {
00221       // Consume open brace.
00222       st.Next();
00223     } else {
00224       if (st.PeekPrev() != "{") {
00225         ostringstream err_ss;
00226         err_ss << "Initializer<vector<T>>: "
00227                << "error: expected '{' at stream position "
00228                << st.PeekPrevTokenStart() << " but found \""
00229                << st.PeekPrev() << "\"";
00230         throw std::runtime_error(err_ss.str());
00231       }
00232     }
00233     while (st.Peek() != "}") {
00234       T vector_element;
00235       Initializer<T> *element_init = new Initializer<T>(&vector_element);
00236       element_init->Init(st, env);
00237       member_->push_back(vector_element);
00238       delete element_init;
00239       // Each vector element initializer must be followed by a comma
00240       // or the final closing parenthesis.
00241       if (st.Peek() != ","  && st.Peek() != "}") {
00242         ostringstream err_ss;
00243         err_ss << "Initializer<vector<T>>: "
00244                << "error: expected ',' or '}' at stream position "
00245                << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
00246         throw std::runtime_error(err_ss.str());
00247       }
00248       // Read comma, if present.
00249       if (st.Peek() == ",") {
00250         st.Next();
00251       }
00252     }
00253     // Consume close brace.
00254     st.Next();
00255   }
