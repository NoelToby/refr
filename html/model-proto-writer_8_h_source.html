00160   virtual void Do(Model *model) {
00161     if (model->best_model_epoch() == prev_best_epoch_index_) {
00162       return;
00163     }
00164     if (verbose_) {
00165       cerr << "Writing out model from epoch " << model->best_model_epoch()
00166            << " to file \"" << model_file_ << "\"...";
00167       cerr.flush();
00168     }
00169     // First, serialize Model to a ModelMessage.
00170     confusion_learning::ModelMessage model_message;
00171     writer_->Write(model, &model_message);
00172     // Next, write out ModelMessage to file.
00173     string tmp_file = model_file_ + ".tmp";
00174     ConfusionProtoIO proto_writer(tmp_file, ConfusionProtoIO::WRITE,
00175                                   compressed_, use_base64_);
00176     proto_writer.Write(model_message);
00177     proto_writer.Close();
00178     rename(tmp_file.c_str(), model_file_.c_str());
00179     // Record which model epoch we've written out.
00180     prev_best_epoch_index_ = model->best_model_epoch();
00181     if (verbose_) {
00182       cerr << " done." << endl;
00183     }
00184   }
