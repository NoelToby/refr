00168   virtual void Do(Model *model) {
00169     if (model->best_model_epoch() == prev_best_epoch_index_) {
00170       return;
00171     }
00172     if (verbose_) {
00173       cerr << "Writing out model from epoch " << model->best_model_epoch()
00174            << " to file \"" << model_file_ << "\"...";
00175       cerr.flush();
00176     }
00177     // First, serialize Model to a ModelMessage.
00178     confusion_learning::ModelMessage model_message;
00179     writer_->Write(model, &model_message);
00180     // Next, write out ModelMessage to file.
00181     string tmp_file = model_file_ + ".tmp";
00182     ConfusionProtoIO proto_writer(tmp_file, ConfusionProtoIO::WRITE,
00183                                   compressed_, use_base64_);
00184     proto_writer.Write(model_message);
00185     proto_writer.Close();
00186     rename(tmp_file.c_str(), model_file_.c_str());
00187     // Record which model epoch we've written out.
00188     prev_best_epoch_index_ = model->best_model_epoch();
00189     if (verbose_) {
00190       cerr << " done." << endl;
00191     }
00192   }
00193  private:
00194   shared_ptr<ModelProtoWriter> writer_;
