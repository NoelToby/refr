00255   virtual void Init(StreamTokenizer &st) {
00256     // Either the next token is an open brace (if reading tokens from
00257     // within a Factory-constructible object's member init list), or
00258     // else we just read an open brace (if Interpreter is reading tokens).
00259     if (st.Peek() == "{") {
00260       // Consume open brace.
00261       st.Next();
00262     } else {
00263       if (st.PeekPrev() != "{") {
00264         ostringstream err_ss;
00265         err_ss << "Initializer<vector<T>>: "
00266                << "error: expected '{' at stream position "
00267                << st.PeekPrevTokenStart() << " but found \""
00268                << st.PeekPrev() << "\"";
00269         throw std::runtime_error(err_ss.str());
00270       }
00271     }
00272     while (st.Peek() != "}") {
00273       T vector_element;
00274       Initializer<T> *element_init = new Initializer<T>("", &vector_element);
00275       element_init->Init(st);
00276       member_->push_back(vector_element);
00277       delete element_init;
00278       // Each vector element initializer must be followed by a comma
00279       // or the final closing parenthesis.
00280       if (st.Peek() != ","  && st.Peek() != "}") {

00283                << "error: expected ',' or '}' at stream position "
00284                << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
00285         throw std::runtime_error(err_ss.str());
00286       }
00287       // Read comma, if present.
00288       if (st.Peek() == ",") {
00289         st.Next();

00291     }
