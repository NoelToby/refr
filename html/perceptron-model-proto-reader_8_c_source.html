00053 void
00054 PerceptronModelProtoReader::Read(const ModelMessage &model_message,
00055                                  Model *model) const {
00056   PerceptronModel *perceptron_model = static_cast<PerceptronModel *>(model);
00057   perceptron_model->name_ = model_message.identifier();
00058   perceptron_model->best_model_epoch_ = model_message.num_iterations();
00059   perceptron_model->time_ = Time(perceptron_model->best_model_epoch_, -1, -1);
00060   // TODO(dbikel): Emit warning if model_message.has_symbols() returns true
00061   //               when perceptron_model->symbols_ is NULL?
00062   if (perceptron_model->symbols_ != NULL && model_message.has_symbols()) {
00063     const SymbolTableMessage &symbol_table_message = model_message.symbols();
00064     for (int i = 0; i < symbol_table_message.symbol_size(); ++i) {
00065       const SymbolMessage &symbol_message = symbol_table_message.symbol(i);
00066       perceptron_model->symbols_->SetIndex(symbol_message.symbol(),
00067                                            symbol_message.index());
00068     }
00069   }
00070   // TODO(dbikel): De-serialize model loss.
00071   if (model_message.has_raw_parameters()) {
00072     fv_reader_.Read(model_message.raw_parameters(),
00073                     perceptron_model->best_models_.weights_,
00074                     perceptron_model->symbols());
00075   }
00076   if (model_message.has_avg_parameters()) {
00077     fv_reader_.Read(model_message.avg_parameters(),
00078                     perceptron_model->best_models_.average_weights_,
00079                     perceptron_model->symbols());
00080   }
00081   // Do "smart copying".
00082   if (smart_copy_) {
00083     if (perceptron_model->best_models_.weights_.size() == 0 &&
00084         perceptron_model->best_models_.average_weights_.size() > 0) {
00085       perceptron_model->best_models_.weights_ =
00086           perceptron_model->best_models_.average_weights_;
00087     } else if (perceptron_model->best_models_.average_weights_.size() == 0 &&
00088                perceptron_model->best_models_.weights_.size() > 0) {
00089       perceptron_model->best_models_.average_weights_ = 
00090           perceptron_model->best_models_.weights_;
00091     }
00092   }
00093 
00094   // Finally, make sure best_models_ is copied to models_.
00095   perceptron_model->models_ = perceptron_model->best_models_;
00096 }

00098 void PerceptronModelProtoReader::ReadFeatures(istream& is,
00099                                               Model *model,
00100                                               bool skip_key,
00101                                               const string& separator) const {
00102   PerceptronModel *perceptron_model = dynamic_cast<PerceptronModel *>(model);
00103   TrainingVectorSet &features = perceptron_model->best_models_;
00104   Symbols *symbols = perceptron_model->symbols();
00105   ConfusionProtoIO proto_reader;
00106   string buffer;
00107   while (is && is.good()) {
00108     getline(is, buffer);
00109     if (buffer.empty()) {
00110       break;
00111     }
00112     if (skip_key) {
00113       size_t seppos = buffer.find(separator);
00114       if (seppos != string::npos) {
00115         buffer.erase(0, seppos+1);
00116       }
00117     }
00118     FeatureMessage feature_msg;
00119     if (!proto_reader.DecodeBase64(buffer, &feature_msg)) {
00120       cerr << "Error decoding: " << feature_msg.Utf8DebugString() << endl;
00121       continue;
00122     }
00123     int uid = feature_msg.id();
00124     if (symbols != NULL &&
00125         feature_msg.has_name() && !feature_msg.name().empty()) {
00126       uid = symbols->GetIndex(feature_msg.name());
00127     }
00128     double value = feature_msg.value();
00129     if (isnan(value)) {
00130         cerr << "PerceptronModelProtoReader: WARNING: feature "
00131              << uid << " has value that is NaN" << endl;
00132     } else {
00133       features.weights_.IncrementWeight(uid, value);
00134     }
00135     if (feature_msg.has_avg_value()) {
00136       double avg_value = feature_msg.avg_value();
00137       if (isnan(avg_value)) {
00138         cerr << "PerceptronModelProtoReader: WARNING: feature "
00139              << uid << " has avg_value that is NaN" << endl;
00140       } else {
00141         features.average_weights_.IncrementWeight(uid, avg_value);
00142       }
00143     }
00144   }
00145   // Do "smart copying".
00146   if (smart_copy_) {
00147     if (features.weights_.size() == 0 && features.average_weights_.size() > 0) {
00148       features.weights_ = features.average_weights_;
00149     } else if (features.average_weights_.size() == 0 &&
00150                features.weights_.size() > 0) {
00151       features.average_weights_ = features.weights_;
00152     }
00153   }
00154   // Make sure to copy latest model to models_.
00155   perceptron_model->models_ = features;
00156 }
