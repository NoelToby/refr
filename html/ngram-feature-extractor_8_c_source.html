00044 void
00045 NgramExtractor::Extract(const vector<string> &tokens,
00046                         const int n,
00047                         const string &prefix,
00048                         FeatureVector<string,double> &symbolic_features) const {
00049   int tokens_len = (int)tokens.size();
00050   int last_token_index = tokens_len - 1;
00051   for (int i = 1; i < tokens_len; ++i) {
00052     int max_prev = i;
00053     for (int prev_index = ((i - n + 1) < 0) ? 0 : (i - n + 1);
00054          prev_index <= max_prev;
00055          ++prev_index) {
00056       // No need to output a feature consisting solely of "</s>" token.
00057       if (max_prev == last_token_index && max_prev == prev_index) {
00058         break;
00059       }
00060       stringstream symbol_ss;
00061       if (prefix.empty()) {
00062         symbol_ss << n << "g_ng{";
00063       } else {
00064         symbol_ss << prefix << "{";
00065       }
00066       for (int j = prev_index; j <= max_prev; ++j) {
00067         symbol_ss << tokens[j] << ((j < max_prev) ? "," : "}");
00068       }
00069       symbolic_features.IncrementWeight(symbol_ss.str(), 1.0);
00070     }
00071   }
00072 }
