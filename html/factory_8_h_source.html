00567     if (token_type == StreamTokenizer::RESERVED_WORD &&
00568         (st.Peek() == "nullptr" || st.Peek() == "NULL")) {
00569       return shared_ptr<T>();
00570     }
00571     if (token_type != StreamTokenizer::IDENTIFIER) {
00572       ostringstream err_ss;

00574              << "error: expected type specifier token but found "
00575              << StreamTokenizer::TypeName(token_type);
00576       throw std::runtime_error(err_ss.str());

00579     // Read the concrete type of object to be created.
00580     string type = st.Next();
00581 
00582     // Read the open parenthesis token.
00583     if (st.Peek() != "(") {
00584       ostringstream err_ss;
00585       err_ss << "Factory<" << BaseName() << ">: "
00586              << "error: expected '(' at stream position "
00587              << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
00588       throw std::runtime_error(err_ss.str());
00589     }
00590     st.Next();
00591 
00592     // Attempt to create an instance of type.
00593     typename unordered_map<string, const Constructor<T> *>::iterator cons_it =
00594         cons_table_->find(type);

00596       ostringstream err_ss;
00597       err_ss << "Factory<" << BaseName() << ">: "
00598              << "error: unknown type: \"" << type << "\"";
00599       throw std::runtime_error(err_ss.str());
00600     }

00602 
00603     // Ask new instance to set up member initializers.
00604     Initializers initializers;
00605     instance->RegisterInitializers(initializers);
00606 
00607     // Parse initializer list.
00608     while (st.Peek() != ")") {
00609       token_type = st.PeekTokenType();
00610       if (token_type != StreamTokenizer::IDENTIFIER) {
00611         ostringstream err_ss;
00612         err_ss << "Factory<" << BaseName() << ">: "
00613                << "error: expected token of type IDENTIFIER at "
00614                << "stream position " << st.PeekTokenStart() << " but found "
00615                << StreamTokenizer::TypeName(token_type) << ": \""
00616                << st.Peek() << "\"";
00617         throw std::runtime_error(err_ss.str());

00619       size_t member_name_start = st.PeekTokenStart();
00620       string member_name = st.Next();
00621       typename Initializers::iterator init_it = initializers.find(member_name);
00622       if (init_it == initializers.end()) {
00623         ostringstream err_ss;
00624         err_ss << "Factory<" << BaseName() << ">: "
00625                << "error: unknown member name \"" << member_name
00626                << "\" in initializer list for type " << type << " at stream "
00627                << "position " << member_name_start;
00628         throw std::runtime_error(err_ss.str());
00629       }
00630       MemberInitializer *member_initializer = init_it->second;
00631 
00632       // Read open parenthesis.
00633       if (st.Peek() != "(") {
00634         ostringstream err_ss;
00635         err_ss << "Factory<" << BaseName() << ">: "
00636                << "error: expected '(' at stream position "
00637                << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
00638         throw std::runtime_error(err_ss.str());
00639       }
00640       st.Next();

00643       member_initializer->Init(st, env_ptr.get());

00645       // Read close parenthesis for current member initializer.
00646       if (st.Peek() != ")") {
00647         ostringstream err_ss;
00648         err_ss << "Factory<" << BaseName() << ">: "
00649                << "error: expected ')' at stream position "
00650                << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
00651         throw std::runtime_error(err_ss.str());

00654

00656       // closing parenthesis.
00657       if (st.Peek() != ","  && st.Peek() != ")") {
00658         ostringstream err_ss;

00660                << "error: expected ',' or ')' at stream position "
00661                << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";
00662         throw std::runtime_error(err_ss.str());
00663       }

00669 
00670     // Read the close parenthesis token for this factory type specification.
00671     if (st.Peek() != ")") {
00672       ostringstream err_ss;
00673       err_ss << "Factory<" << BaseName() << ">: "
00674              << "error: expected ')' at stream position "
00675              << st.PeekTokenStart() << " but found \"" << st.Peek() << "\"";

00678     st.Next();
00679 
00680     // Run through all member initializers: if any are required but haven't
00681     // been invoked, it is an error.

00683          init_it != initializers.end();
