00065 bool
00066 StreamTokenizer::GetNext(Token *next) {
00067   if (!is_.good()) {
00068     eof_reached_ = true;
00069     return false;
00070   }
00071   // Get first character of next token.
00072   char c;
00073   bool is_whitespace = true;
00074   while (is_whitespace) {
00075     if (!ReadChar(&c)) {
00076       return false;
00077     }
00078     is_whitespace = isspace(c);
00079 
00080     // If we find a comment character, then read to the end of the line.
00081     if (!is_whitespace && c == '/' && is_.peek() == '/') {
00082       while (c != '\n') {
00083         if (!ReadChar(&c)) {
00084           return false;
00085         }
00086       }
00087       is_whitespace = true;
00088     }
00089   }
00090 
00091   // In case we're successful in reading the next token, we fill in
00092   // the two stream state data available now.
00093   next->start = num_read_ - 1;
00094   next->line_number = line_number_;
00095 
00096   bool next_tok_complete = false;
00097   next->tok.clear();
00098   if (ReservedChar(c)) {
00099     next->tok += c;
00100     next_tok_complete = true;
00101     next->type = RESERVED_CHAR;
00102   } else if (c == '"') {
00103     // We've got a string literal, so keep reading characters,
00104     // until hitting a non-escaped double quote.
00105     streampos string_literal_start_pos = num_read_ - 1;
00106     bool found_closing_quote = false;
00107     while (is_.good()) {
00108       bool success = ReadChar(&c);
00109       if (success) {
00110         if (c == '"') {
00111           found_closing_quote = true;
00112           break;
00113         } else if (c == '\\') {
00114           success = ReadChar(&c);
00115         }
00116       }
00117       if (success) {
00118         next->tok += c;
00119       }
00120     }
00121     if (!found_closing_quote) {
00122       cerr << "StreamTokenizer: error: could not find closing "
00123            << "double quote for string literal beginning at stream index "
00124            << string_literal_start_pos
00125            << "; partial string literal read: \""
00126            << next->tok << endl;
00127       throw std::runtime_error("unclosed string literal");
00128     }
00129     next_tok_complete = true;
00130     next->type = STRING;
00131   } else {
00132     // This is a number, a reserved word or C++ identifier token, so
00133     // add first character; the remainder of the token will be handled
00134     // in the next block.
00135     next->tok += c;
00136     next->type = (c == '-' || (c >= '0' && c <= '9')) ? NUMBER : IDENTIFIER;
00137   }
00138   if (!next_tok_complete) {
00139     // The current token is a number, a reserved word or C++
00140     // identifier, so we keep reading characters until hitting a
00141     // "reserved character", a whitespace character or EOF.
00142     bool done = false;
00143     while (!done && is_.good()) {
00144       // We don't call ReadChar below because the next character might
00145       // tell us that the current token has ended (i.e., if it's a
00146       // reserved character, a double quote or a whitespace
00147       // character).
00148       int peek = is_.peek();
00149       if (peek != EOF) {
00150         char next_char = static_cast<char>(peek);
00151         if (ReservedChar(next_char) || next_char == '"' || isspace(next_char)) {
00152           // Now that we've finished reading something that is not a
00153           // string literal, change its type to be RESERVED_WORD if it
00154           // exactly matches something in the set of reserved words.
00155           if (reserved_words_.count(next->tok) != 0) {
00156             next->type = RESERVED_WORD;
00157           }
00158           done = true;
00159         } else {
00160           ReadChar(&c);
00161           next->tok += c;
00162         }
00163       } else {
00164         eof_reached_ = true;
00165       }
00166     }
00167   }
00168   // We're about to return a successfully read token, so we make sure to record
00169   // the stream position at this point in the Token object.
00170   next->curr_pos = num_read_;
00171 
00172   return true;
00173 }
