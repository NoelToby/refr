00060 void
00061 DefaultCandidateSetScorer::Score(Model *model,
00062                                  CandidateSet &candidates, bool training) {
00063   // N.B.: We assume there is at least one candidate!
00064   CandidateSet::iterator it = candidates.begin();
00065 
00066   // Score first candidate, which is perforce the best candidate so far.
00067   model->ScoreCandidate(*(*it), training);
00068 
00069   CandidateSet::iterator best_it = it;
00070   CandidateSet::iterator gold_it = it;
00071   ++it;
00072 
00073   // Score any and all remaining candidates.
00074   for ( ; it != candidates.end(); ++it) {
00075     Candidate &candidate = *(*it);
00076     model->ScoreCandidate(candidate, training);
00077     if (model->score_comparator()->Compare(*model, candidate, **best_it) > 0) {
00078       best_it = it;
00079     }
00080     if (model->gold_comparator()->Compare(*model, candidate, **gold_it) > 0) {
00081       gold_it = it;
00082     }
00083   }
00084   candidates.set_best_scoring_index((*best_it)->index());
00085   candidates.set_gold_index((*gold_it)->index());
00086 }

00091 size_t
00092 RandomPairCandidateSetScorer::GetRandomIndex(size_t max) {
00093   // Get an index proportional to the reciprocal rank.
00094   // First, get a floating-point value that is distributed uniformly (roughly)
00096   double r = rand() / (double)RAND_MAX;
00097   // We compute the cummulative density function, or cdf, of the
00098   // reciprocal rank distribution over the fixed set of items.  As soon as
00099   // our uniformly-distributed r is less than the cdf at index i, we
00100   // return that index.
00101   double cdf = 0.0;
00102   double denominator = (max * (max + 1)) / 2;
00103   for (size_t i = 0; i < max; ++i) {
00104     cdf += (max - i) / denominator;
00105     if (r <= cdf) {
00106       return i;
00107     }
00108   }
00109   return max - 1;
00110 }

00112 void
00113 RandomPairCandidateSetScorer::Score(Model *model,
00114                                     CandidateSet &candidates, bool training) {
00115   // First, pick two candidate indices at random.
00116   size_t idx1 = GetRandomIndex(candidates.size());
00117   size_t idx2 = GetRandomIndex(candidates.size());
00118   Candidate &c1 = candidates.Get(idx1);
00119   Candidate &c2 = candidates.Get(idx2);
00120 
00121   // Next, just score those two candidates.
00122   model->ScoreCandidate(c1, training);
00123   model->ScoreCandidate(c2, training);
00124 
00125   // Finally, set indices of best scoring and gold amongst just those two.
00126   int score_cmp = model->score_comparator()->Compare(*model, c1, c2);
00127   candidates.set_best_scoring_index(score_cmp > 0 ? c1.index() : c2.index());
00128 
00129   int gold_cmp = model->gold_comparator()->Compare(*model, c1, c2);
00130   candidates.set_gold_index(gold_cmp > 0 ? c1.index() : c2.index());
00131 }
