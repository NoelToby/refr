00156                         FeatureVector<int, V> &vector) {
00157     FeatureVector<int, V> old_vector = vector;

00177   void UpdateAverage(const Time &time, int uid) {
00178     int iterations_since_update =
00179         time.absolute_index() - last_update_indices_.GetValue(uid);
00180     if (iterations_since_update <= 0) {
00181       return;
00182     }
00183     // Add in perceptron values to weight sum.
00184     double add_to_sum = iterations_since_update * weights_.GetWeight(uid);
00185     double new_weight_sum = weight_sums_.IncrementWeight(uid, add_to_sum);
00186     average_weights_.SetWeight(uid, new_weight_sum / time.absolute_index());
00187     last_update_indices_.SetValue(uid, time.absolute_index());
00188   }
