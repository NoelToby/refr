00049         candidate_set_message->add_candidate();

00051     if (candidate.raw_data() != "") {
00052       candidate_message->set_raw_data(candidate.raw_data());
00053     }
00054 
00055     // Add loss.
00056     ScoreMessage *loss_message = candidate_message->add_score();
00057     loss_message->set_type(ScoreMessage::LOSS);
00058     loss_message->set_score(candidate.loss());
00059     // Add model score.
00060     ScoreMessage *model_score_message = candidate_message->add_score();
00061     model_score_message->set_type(ScoreMessage::OUTPUT_SCORE);
00062     model_score_message->set_score(candidate.score());
00063     // Add baseline score.
00064     ScoreMessage *baseline_score_message = candidate_message->add_score();
00065     baseline_score_message->set_type(ScoreMessage::SYSTEM_SCORE);
00066     baseline_score_message->set_score(candidate.baseline_score());
00067 
00068     // Add features.
00069     FeatureVecMessage *fv_message = candidate_message->mutable_feats();
00070     fv_writer_.Write(candidate.features(), FeatureMessage::BASIC, fv_message);
00071     symbolic_fv_writer_.Write(candidate.symbolic_features(),
00072                               FeatureMessage::BASIC, fv_message);
00073   }
00074 
00075   // Add gold index and best-scoring indices.
