00040 Environment::Environment(int debug) {
00041   debug_ = debug;
00042 
00043   // Set up VarMap instances for each of the primitive types and their vectors.
00044   var_map_["bool"] = new VarMap<bool>("bool");
00045   var_map_["int"] = new VarMap<int>("int");
00046   var_map_["double"] = new VarMap<double>("double");
00047   var_map_["string"] = new VarMap<string>("string");
00048   var_map_["bool[]"] = new VarMap<vector<bool> >("bool[]");
00049   var_map_["int[]"] = new VarMap<vector<int> >("int[]");
00050   var_map_["double[]"] = new VarMap<vector<double> >("double[]");
00051   var_map_["string[]"] = new VarMap<vector<string> >("string[]");
00052 
00053   // Set up VarMap instances for each of the Factory-constructible types
00054   // and their vectors.
00055   for (FactoryContainer::iterator factory_it = FactoryContainer::begin();
00056        factory_it != FactoryContainer::end(); ++factory_it) {
00057     unordered_set<string> registered;
00058     (*factory_it)->CollectRegistered(registered);
00059     string base_name = (*factory_it)->BaseName();
00060 
00061     // Create type-specific VarMap from the Factory and add to var_map_.
00062     VarMapBase *obj_var_map = (*factory_it)->CreateVarMap();
00063     var_map_[obj_var_map->Name()] = obj_var_map;
00064 
00065     if (debug_ >= 2) {
00066       cerr << "Environment: created VarMap for " << obj_var_map->Name()
00067            << endl;
00068     }
00069 
00070     // Create VarMap for vectors of shared_object of T and add to var_map_.
00071     VarMapBase *obj_vector_var_map = (*factory_it)->CreateVectorVarMap();
00072     var_map_[obj_vector_var_map->Name()] = obj_vector_var_map;
00073 
00074     if (debug_ >= 2) {
00075       cerr << "Environment: created VarMap for " << obj_vector_var_map->Name()
00076            << endl;
00077     }
00078 
00079     for (unordered_set<string>::const_iterator it = registered.begin();
00080          it != registered.end(); ++it) {
00081       const string &concrete_type_name = *it;
00082 
00083       unordered_map<string, string>::const_iterator concrete_to_factory_it =
00084           concrete_to_factory_type_.find(concrete_type_name);
00085       if (concrete_to_factory_it != concrete_to_factory_type_.end()) {
00086         // Warn user that there are two entries for the same conrete type

00089              << "concrete-to-factory type mapping ["
00090              << concrete_type_name << " --> " << concrete_to_factory_it->second
00091              << "] with [" << concrete_type_name << " --> " << base_name

00095 
00096       if (debug_ >= 2) {
00097         cerr << "Environment: associating concrete typename "
00098              << concrete_type_name

00100       }
00101     }

00105 void
00106 Environment::ReadAndSet(const string &varname, StreamTokenizer &st) {
00107   bool is_vector =
00108       st.PeekTokenType() == StreamTokenizer::RESERVED_CHAR &&
00109       st.Peek() == "{";
00110   if (is_vector) {
00111     // Consume open brace.
00112     st.Next();
00113   } else if (st.PeekTokenType() == StreamTokenizer::RESERVED_CHAR ||
00114              (st.PeekTokenType() == StreamTokenizer::RESERVED_WORD &&
00115               st.Peek() != "true" && st.Peek() != "false")) {
00116     ostringstream err_ss;
00117     err_ss << "Environment: error: expected type but found token \""
00118            << st.Peek() << "\" of type "
00119            << StreamTokenizer::TypeName(st.PeekTokenType());
00120     throw std::runtime_error(err_ss.str());
00121   }
00122   string next_tok = st.Peek();
00123   bool is_object_type = false;
00124 
00125   string type = InferType(st, is_vector, &is_object_type);
00126 
00127   if (is_object_type) {
00128     // Verify that next_tok is a concrete typename.
00129     unordered_map<string, string>::const_iterator it =
00130         concrete_to_factory_type_.find(next_tok);
00131     if (it == concrete_to_factory_type_.end()) {
00132       ostringstream err_ss;
00133       err_ss << "Environment: error: variable "
00134              << varname << " appears to be of type " << type
00135              << " but token " << next_tok
00136              << " is not a concrete object typename";
00137       throw std::runtime_error(err_ss.str());
00138     }
00139 
00140     // Set type to be abstract factory type.
00141     if (debug_ >= 1) {
00142       cerr << "Environment::ReadAndSet: concrete type is " << type
00143            << "; mapping to abstract Factory type " << it->second << endl;
00144     }
00145     type = it->second;
00146   }
00147 
00148   if (debug_ >= 1) {
00149     cerr << "Environment::ReadAndSet: "
00150          << "next_tok=" << next_tok << "; type=" << type << endl;
00151   }
00152 
00153 
00154   if (type == "") {
00155     ostringstream err_ss;
00156     err_ss << "Environment: error: could not infer type for variable "
00157            << varname;
00158     throw std::runtime_error(err_ss.str());
00159   }
00160 
00161   // Check that type is a key in var_map_.
00162 
00163   var_map_[type]->ReadAndSet(varname, st);
00164   types_[varname] = type;
00165 }

00167 string
00168 Environment::InferType(const StreamTokenizer &st, bool is_vector,
00169                        bool *is_object_type) {
00170   *is_object_type = false;
00171   string next_tok = st.Peek();
00172   switch (st.PeekTokenType()) {
00173     case StreamTokenizer::RESERVED_WORD:
00174       if (next_tok == "true" || next_tok == "false") {
00175         return is_vector ? "bool[]" : "bool";
00176       } else {
00177         return "";
00178       }
00179       break;
00180     case StreamTokenizer::STRING:
00181       return is_vector ? "string[]" : "string";
00182       break;
00183     case StreamTokenizer::NUMBER:
00184       {
00185         // If a token is a NUMBER, it is a double iff it contains a
00186         // decimal point.
00187         size_t dot_pos = next_tok.find('.');
00188         if (dot_pos != string::npos) {
00189           return is_vector ? "double[]" : "double";
00190         } else {
00191           return is_vector ? "int[]" : "int";
00192         }
00193       }
00194       break;
00195     case StreamTokenizer::IDENTIFIER:
00196       {
00197         *is_object_type = true;
00198         string type = is_vector ? next_tok + "[]" : next_tok;
00199         return type;
00200       }
00201       break;
00202     default:
00203       return "";
00204   }
00205 }
