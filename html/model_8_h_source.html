00061  public:
00066   virtual int Compare(const Model &model,
00067                       const Candidate &c1, const Candidate &c2) {
00068     double score_diff = c1.score() - c2.score();

00075 class DefaultGoldComparator : public Candidate::Comparator {
00076  public:
00085   virtual int Compare(const Model &model,
00086                       const Candidate &c1, const Candidate &c2) {
00087     double loss_diff = c1.loss() - c2.loss();
00088     double score_diff = c1.score() - c2.score();
00089     // If c1 has "less loss" it is "better", so we return 1.
00090     if (loss_diff < 0.0) {
00091       return 1;
00092     } else if (loss_diff > 0.0) {
00093       return -1;
00094     } else {
00095       // If c1&rsquo;s and c2&rsquo;s losses are equal, we compare
00096       // them based on score, where a higher score is &ldquo;better&rdquo;
00097       // (meaning a positive return value).
00098       return score_diff == 0.0 ? 0 : (score_diff < 0.0 ? -1 : 1);
00099     }
00100   }
00101 };
