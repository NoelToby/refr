00041 EnvironmentImpl::EnvironmentImpl(int debug) {
00042   debug_ = debug;
00043 
00044   // Set up VarMap instances for each of the primitive types and their vectors.
00045   var_map_["bool"] = new VarMap<bool>("bool", this);
00046   var_map_["int"] = new VarMap<int>("int", this);
00047   var_map_["double"] = new VarMap<double>("double", this);
00048   var_map_["string"] = new VarMap<string>("string", this);
00049   var_map_["bool[]"] = new VarMap<vector<bool> >("bool[]", this);
00050   var_map_["int[]"] = new VarMap<vector<int> >("int[]", this);
00051   var_map_["double[]"] = new VarMap<vector<double> >("double[]", this);
00052   var_map_["string[]"] = new VarMap<vector<string> >("string[]", this);
00053 
00054   // Set up VarMap instances for each of the Factory-constructible types
00055   // and their vectors.
00056   for (FactoryContainer::iterator factory_it = FactoryContainer::begin();
00057        factory_it != FactoryContainer::end(); ++factory_it) {
00058     unordered_set<string> registered;
00059     (*factory_it)->CollectRegistered(registered);
00060     string base_name = (*factory_it)->BaseName();
00061 
00062     // Create type-specific VarMap from the Factory and add to var_map_.
00063     VarMapBase *obj_var_map = (*factory_it)->CreateVarMap(this);
00064     var_map_[obj_var_map->Name()] = obj_var_map;
00065 
00066     if (debug_ >= 2) {
00067       cerr << "Environment: created VarMap for " << obj_var_map->Name()
00068            << endl;
00069     }
00070 
00071     // Create VarMap for vectors of shared_object of T and add to var_map_.
00072     VarMapBase *obj_vector_var_map = (*factory_it)->CreateVectorVarMap(this);
00073     var_map_[obj_vector_var_map->Name()] = obj_vector_var_map;
00074 
00075     if (debug_ >= 2) {
00076       cerr << "Environment: created VarMap for " << obj_vector_var_map->Name()
00077            << endl;
00078     }
00079 
00080     for (unordered_set<string>::const_iterator it = registered.begin();
00081          it != registered.end(); ++it) {
00082       const string &concrete_type_name = *it;
00083 
00084       unordered_map<string, string>::const_iterator concrete_to_factory_it =
00085           concrete_to_factory_type_.find(concrete_type_name);
00086       if (concrete_to_factory_it != concrete_to_factory_type_.end()) {
00087         // Warn user that there are two entries for the same concrete type
00088         // (presumably to different abstract factory types).
00089         cerr << "Environment: WARNING: trying to override existing "
00090              << "concrete-to-factory type mapping ["
00091              << concrete_type_name << " --> " << concrete_to_factory_it->second
00092              << "] with [" << concrete_type_name << " --> " << base_name
00093              << endl;
00094       }
00095       concrete_to_factory_type_[concrete_type_name] = base_name;
00096 
00097       if (debug_ >= 2) {
00098         cerr << "Environment: associating concrete typename "
00099              << concrete_type_name
00100              << " with factory for " << base_name << endl;
00101       }
00102     }
00103   }

00106 void
00107 EnvironmentImpl::ReadAndSet(const string &varname, StreamTokenizer &st,
00108                             const string type) {
00109   bool is_vector =
00110       st.PeekTokenType() == StreamTokenizer::RESERVED_CHAR &&
00111       st.Peek() == "{";
00112 
00113   if (is_vector) {
00114     // Consume open brace.
00115     st.Next();
00116   } else if (st.PeekTokenType() == StreamTokenizer::RESERVED_CHAR ||
00117              (st.PeekTokenType() == StreamTokenizer::RESERVED_WORD &&
00118               st.Peek() != "true" && st.Peek() != "false")) {
00119     ostringstream err_ss;
00120     err_ss << "Environment: error: expected type but found token \""
00121            << st.Peek() << "\" of type "
00122            << StreamTokenizer::TypeName(st.PeekTokenType());
00123     throw std::runtime_error(err_ss.str());
00124   }
00125 
00126   string next_tok = st.Peek();
00127   bool is_object_type = false;
00128 
00129   string inferred_type = InferType(varname, st, is_vector, &is_object_type);
00130 
00131   if (debug_ >= 1) {
00132     cerr << "Environment::ReadAndSet: next_tok=\"" << next_tok
00133          << "\"; explicit type=\"" << type << "\"; "
00134          << "inferred_type=\"" << inferred_type << "\"" << endl;
00135   }
00136 
00137   if (type == "" && inferred_type == "") {
00138     ostringstream err_ss;
00139     err_ss << "Environment: error: no explicit type specifier and could not "
00140            << "infer type for variable " << varname;
00141     throw std::runtime_error(err_ss.str());
00142   }
00143   if (type != "" && inferred_type != "" && type != inferred_type) {
00144     ostringstream err_ss;
00145     err_ss << "Environment: error: explicit type " << type
00146            << " and inferred type " << inferred_type
00147            << " disagree for variable " << varname;
00148     throw std::runtime_error(err_ss.str());
00149   }
00150 
00151   // If no explicit type specifier, then the inferred_type is the type.
00152   string varmap_type = type == "" ? inferred_type : type;
00153 
00154   // Check that varmap_type is a key in var_map_.
00155   var_map_[varmap_type]->ReadAndSet(varname, st);
00156   types_[varname] = varmap_type;
00157 }

00159 string
00160 EnvironmentImpl::InferType(const string &varname,
00161                            const StreamTokenizer &st, bool is_vector,
00162                            bool *is_object_type) {
00163   *is_object_type = false;
00164   string next_tok = st.Peek();
00165   switch (st.PeekTokenType()) {
00166     case StreamTokenizer::RESERVED_WORD:
00167       if (next_tok == "true" || next_tok == "false") {
00168         return is_vector ? "bool[]" : "bool";
00169       } else {
00170         return "";
00171       }
00172       break;
00173     case StreamTokenizer::STRING:
00174       return is_vector ? "string[]" : "string";
00175       break;
00176     case StreamTokenizer::NUMBER:
00177       {
00178         // If a token is a NUMBER, it is a double iff it contains a
00179         // decimal point.
00180         size_t dot_pos = next_tok.find('.');
00181         if (dot_pos != string::npos) {
00182           return is_vector ? "double[]" : "double";
00183         } else {
00184           return is_vector ? "int[]" : "int";
00185         }
00186       }
00187       break;
00188     case StreamTokenizer::IDENTIFIER:
00189       {
00190         string type = "";
00191 
00192         // Find out if next_tok is a concrete typename or a variable.
00193         unordered_map<string, string>::const_iterator factory_type_it =
00194             concrete_to_factory_type_.find(next_tok);
00195         unordered_map<string, string>::const_iterator var_type_it =
00196             types_.find(next_tok);
00197         if (factory_type_it != concrete_to_factory_type_.end()) {
00198           // Set type to be abstract factory type.
00199           if (debug_ >= 1) {
00200             cerr << "Environment::InferType: concrete type is " << next_tok
00201                  << "; mapping to abstract Factory type "
00202                  << factory_type_it->second << endl;
00203           }
00204           type = factory_type_it->second;
00205           *is_object_type = true;
00206           type = is_vector ? type + "[]" : type;
00207 
00208           if (debug_ >= 1) {
00209             cerr << "Environment::InferType: type "
00210                  << (is_vector ? "is" : "isn't")
00211                  << " a vector, so final inferred type is " << type << endl;
00212           }
00213         } else if (var_type_it != types_.end()) {
00214           // Could be a variable, in which case we need not only to return
00215           // the variable's type, but also set is_object_type and is_vector
00216           // based on the variable's type string.
00217           string append = is_vector ? "[]" : "";
00218           type = var_type_it->second + append;
00219           if (debug_ >= 1) {
00220             cerr << "Environment::InferType: found variable "
00221                  << var_type_it->first << " of type " << var_type_it->second
00222                  << "; type is " << type << endl;
00223           }
00224         } else {
00225           ostringstream err_ss;
00226           err_ss << "Environment: error: token " << next_tok
00227                  << " is neither a variable nor a concrete object typename";
00228           throw std::runtime_error(err_ss.str());
00229         }
00230         return type;
00231       }
00232       break;
00233     default:
00234       return "";
00235   }
00236 }
