00046 void
00047 Interpreter::Eval(StreamTokenizer &st) {
00048   // Keeps reading assignment statements until there are no more tokens.
00049   while (st.PeekTokenType() != StreamTokenizer::EOF_TYPE) {
00050     try {
00051     // Read variable name or type specifier.
00052     StreamTokenizer::TokenType token_type = st.PeekTokenType();
00053     VarMapBase *varmap = env_->GetVarMapForType(st.Peek());
00054     bool is_type_specifier =  varmap != NULL;
00055     if (token_type != StreamTokenizer::IDENTIFIER && !is_type_specifier) {
00056       string expected_type =
00057           string(StreamTokenizer::TypeName(StreamTokenizer::IDENTIFIER)) +
00058           " or type specifier";
00059       string found_type = StreamTokenizer::TypeName(token_type);
00060       WrongTokenTypeError(st.PeekTokenStart(), expected_type, found_type,
00061                           st.Peek());
00062     }
00063 
00064     string type = "";
00065     if (is_type_specifier) {
00066       // Consume and remember the type specifier.
00067       st.Next();              // Explicit type could be a concrete type.
00068       type = varmap->Name();  // Remember the abstract type.
00069 
00070       // Check that next token is a variable name.
00071       StreamTokenizer::TokenType token_type = st.PeekTokenType();
00072       if (token_type != StreamTokenizer::IDENTIFIER) {
00073         WrongTokenTypeError(st.PeekTokenStart(), StreamTokenizer::IDENTIFIER,
00074                             token_type, st.Peek());
00075       }
00076     }
00077 
00078     string varname = st.Next();
00079 
00080     // Next, read equals sign.
00081     token_type = st.PeekTokenType();
00082     if (st.Peek() != "=") {
00083       WrongTokenError(st.PeekTokenStart(), "=", st.Peek(), st.PeekTokenType());
00084     }
00085 
00086     // Consume equals sign.
00087     st.Next();
00088 
00089     if (st.PeekTokenType() == StreamTokenizer::EOF_TYPE) {
00090       ostringstream err_ss;
00091       err_ss << "Interpreter:" << filename_
00092              << ": error: unexpected EOF at stream position "
00093              << st.tellg();
00094       throw std::runtime_error(err_ss.str());
00095     }
00096 
00097     // Consume and set the value for this variable in the environment.
00098     env_->ReadAndSet(varname, st, type);
00099 
00100     token_type = st.PeekTokenType();
00101     if (st.Peek() != ";") {
00102       WrongTokenError(st.PeekTokenStart(), ";", st.Peek(), st.PeekTokenType());
00103     }
00104     // Consume semicolon.
00105     st.Next();
00106     }
00107     catch (std::runtime_error &e) {
00108       cerr << "threw exception: " << e.what() << endl;
00109       // For now, we simply give up.
00110       break;
00111     }
00112   }
00113 }
